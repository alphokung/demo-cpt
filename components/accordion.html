Basic Accordion

คลิกที่หัวข้อเพื่อขยาย/ย่อเนื้อหา

ใบรับรองแพทย์ดิจิทัลคืออะไร? expand_more
ใบรับรองแพทย์ดิจิทัล คือเอกสารรับรองสุขภาพในรูปแบบอิเล็กทรอนิกส์ ออกโดยแพทย์ผ่านระบบหมอพร้อม สามารถใช้แทนเอกสารกระดาษได้ในหน่วยงานที่เข้าร่วมโครงการ
จะขอใบรับรองแพทย์ดิจิทัลได้อย่างไร? expand_more
สามารถขอรับที่โรงพยาบาลหรือคลินิกที่เข้าร่วมโครงการ โดยแพทย์จะออกใบรับรองผ่านระบบ และเอกสารจะปรากฏในแอปพลิเคชันของท่านภายใน 24 ชั่วโมง
ใบรับรองแพทย์ดิจิทัลหมดอายุเมื่อไร? expand_more
อายุของใบรับรองแพทย์ขึ้นอยู่กับประเภท เช่น ใบรับรองตรวจสุขภาพทั่วไปมักมีอายุ 1 ปี ส่วนใบรับรองสำหรับใบขับขี่มีอายุ 1 ปีเช่นกัน ระบบจะแจ้งเตือนเมื่อใกล้หมดอายุ
หากเอกสารหมดอายุจะต้องทำอย่างไร? expand_more
ติดต่อโรงพยาบาลหรือคลินิกเพื่อตรวจสุขภาพและออกใบรับรองแพทย์ใหม่ ไม่สามารถต่ออายุเอกสารเดิมได้ จะต้องออกเอกสารใหม่เท่านั้น

วิธีใช้งาน

HTML

<div class="accordion">

  <!-- เปิดค่าเริ่มต้น: เพิ่ม class .open -->
  <div class="accordion-item open">
    <div class="accordion-header" onclick="toggleAccordion(this)">
      <span class="accordion-title">หัวข้อ</span>
      <span class="accordion-icon">
        <span class="material-symbols-outlined">expand_more</span>
      </span>
    </div>
    <div class="accordion-body">
      <div class="accordion-body-inner">เนื้อหา...</div>
    </div>
  </div>

  <div class="accordion-item">
    <!-- ...รายการถัดไป -->
  </div>

</div>

JavaScript

function toggleAccordion(header) {
  var item = header.closest('.accordion-item');
  var body = item.querySelector('.accordion-body');
  var isOpen = item.classList.contains('open');

  // ปิดทุกรายการในกลุ่มเดียวกัน (exclusive mode)
  // var accordion = item.closest('.accordion');
  // accordion.querySelectorAll('.accordion-item.open').forEach(function(el) {
  //   el.classList.remove('open');
  //   el.querySelector('.accordion-body').style.maxHeight = null;
  // });

  if (isOpen) {
    item.classList.remove('open');
    body.style.maxHeight = null;
  } else {
    item.classList.add('open');
    body.style.maxHeight = body.scrollHeight + 'px';
  }
}
