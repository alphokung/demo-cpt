CSS custom properties

/* spacing token */
margin-top: var(--spacing-xl);      /* 16px */
padding: var(--spacing-lg);         /* 12px */
gap: var(--spacing-md);             /* 8px */

/* radius token */
border-radius: var(--radius-2xl);   /* 16px — card */
border-radius: var(--radius-full);  /* pill */

Utility classes (HTML)

<!-- margin -->
<div class="mt-xl">...</div>   <!-- margin-top: 16px -->
<div class="mb-lg">...</div>   <!-- margin-bottom: 12px -->

<!-- padding -->
<div class="p-xl">...</div>    <!-- padding: 16px -->
<div class="px-lg py-md">...</div>  <!-- px:12px py:8px -->

<!-- gap (on flex/grid parent) -->
<div style="display:flex" class="gap-md">...</div>
