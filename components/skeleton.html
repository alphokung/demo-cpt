Base Classes

Title — .skeleton-title

Circle — .skeleton-circle

Button — .skeleton-button

วิธีใช้งาน

Core class

<!-- เพิ่ม skeleton-loading ให้ element ใดก็ได้ -->
<div class="skeleton-loading" style="height:16px; width:80%"></div>

<!-- Text preset -->
<div class="skeleton-loading skeleton-text"></div>         <!-- 100% width -->
<div class="skeleton-loading skeleton-text medium"></div>  <!-- 70% -->
<div class="skeleton-loading skeleton-text short"></div>   <!-- 40% -->

<!-- Title -->
<div class="skeleton-loading skeleton-title"></div>

<!-- Circle (avatar) -->
<div class="skeleton-loading skeleton-circle skeleton-avatar"></div>

<!-- Button -->
<div class="skeleton-loading skeleton-button"></div>

Toggle skeleton ↔ loaded via JavaScript

// ซ่อน skeleton, แสดง content จริง
function setLoaded(isLoaded) {
  document.getElementById('skeleton-view').style.display =
    isLoaded ? 'none' : 'block';
  document.getElementById('content-view').style.display =
    isLoaded ? 'block' : 'none';
}

บน dark background — ใช้ rgba แทนสีเทา

<!-- inside .welcome-card / .welcome-banner -->
<span class="skeleton-loading"
  style="background:rgba(255,255,255,0.25) !important;
         width:160px; height:22px; display:block">
</span>
