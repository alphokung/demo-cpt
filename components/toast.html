HTML — container (วางในระดับ body)

<div class="toast-container" id="toast-container"></div>

JavaScript — showToast() function

function showToast(message, type = 'default', icon = 'notifications', duration = 3000) {
  const container = document.getElementById('toast-container');
  const el = document.createElement('div');
  const variantClass = type !== 'default' ? ` toast--${type}` : '';
  el.className = `toast${variantClass}`;
  el.innerHTML = `
    <span class="material-symbols-outlined">${icon}</span>
    <span class="toast__message">${message}</span>
    <button class="toast__close" onclick="dismissToast(this.closest('.toast'))">
      <span class="material-symbols-outlined">close</span>
    </button>`;
  container.appendChild(el);
  setTimeout(() => dismissToast(el), duration);
}

function dismissToast(el) {
  if (!el || el.classList.contains('toast-exit')) return;
  el.classList.add('toast-exit');
  el.addEventListener('animationend', () => el.remove(), { once: true });
}

// ตัวอย่างการเรียกใช้:
showToast('บันทึกสำเร็จ', 'success', 'check_circle');
showToast('เกิดข้อผิดพลาด', 'danger', 'error');
showToast('คำเตือน', 'warning', 'warning');
showToast('ข้อมูล', 'info', 'info');
showToast('ข้อความ', 'default', 'notifications');
showToast('ข้อความ', 'light', 'send');
