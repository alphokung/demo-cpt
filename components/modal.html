Modal & Bottom Sheet

Basic Modal

Modal พื้นฐาน — header, body, footer พร้อมปุ่มปิด

Bottom Sheet

Slide-up panel จากด้านล่าง — เหมาะกับ mobile action sheet

วิธีใช้งาน

HTML — Modal

<!-- Overlay -->
<div class="modal-overlay" id="my-modal">
  <div class="modal-content">

    <!-- Header -->
    <div class="modal-header">
      <h3 class="modal-title">ชื่อ Modal</h3>
      <button class="modal-close-btn" onclick="closeModal('my-modal')">
        <span class="material-symbols-outlined">close</span>
      </button>
    </div>

    <!-- Body -->
    <div class="modal-body">เนื้อหา...</div>

    <!-- Footer -->
    <div class="modal-footer modal-footer--row">
      <button class="btn btn-secondary" onclick="closeModal('my-modal')">ยกเลิก</button>
      <button class="btn btn-primary">ยืนยัน</button>
    </div>

  </div>
</div>

HTML — Bottom Sheet

<!-- เพิ่ม class .bottom-sheet บน .modal-overlay -->
<div class="modal-overlay bottom-sheet" id="my-sheet">
  <div class="modal-content">
    <div class="bottom-sheet-handle"></div>
    <div class="modal-header">...</div>
    <div class="modal-body">...</div>
    <div class="modal-footer">...</div>
  </div>
</div>

JavaScript

function openModal(id) {
  document.getElementById(id).classList.add('active');
}
function closeModal(id) {
  document.getElementById(id).classList.remove('active');
}
// ปิดเมื่อคลิก backdrop
document.querySelectorAll('.modal-overlay').forEach(overlay => {
  overlay.addEventListener('click', e => {
    if (e.target === overlay) closeModal(overlay.id);
  });
});
