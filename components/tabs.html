HTML — Underline Tabs

<div class="tabs" id="my-tabs">
  <div class="tab-bar">
    <button class="tab-btn active"
      onclick="switchTab('my-tabs', 'panel-1', this)">
      แท็บที่ 1
    </button>
    <button class="tab-btn"
      onclick="switchTab('my-tabs', 'panel-2', this)">
      แท็บที่ 2
    </button>
  </div>
  <div class="tab-panels">
    <div class="tab-panel active" id="panel-1">เนื้อหา 1</div>
    <div class="tab-panel"       id="panel-2">เนื้อหา 2</div>
  </div>
</div>

HTML — Pill Tabs (เพิ่ม .tab-bar--pill)

<div class="tab-bar tab-bar--pill">
  <button class="tab-btn active" onclick="switchTab(...)">วันนี้</button>
  <button class="tab-btn" onclick="switchTab(...)">สัปดาห์นี้</button>
</div>

JavaScript

function switchTab(tabsId, panelId, btn) {
  var tabs = document.getElementById(tabsId);
  // ปิดแท็บทั้งหมด
  tabs.querySelectorAll('.tab-btn').forEach(function(b) {
    b.classList.remove('active');
  });
  tabs.querySelectorAll('.tab-panel').forEach(function(p) {
    p.classList.remove('active');
  });
  // เปิดแท็บที่เลือก
  btn.classList.add('active');
  document.getElementById(panelId).classList.add('active');
}
